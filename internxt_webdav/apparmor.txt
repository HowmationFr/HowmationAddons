#include <tunables/global>

profile internxt_webdav flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Broad allows — the upstream Node + Internxt CLI legitimately need them
  # (libc, /app, /root/.internxt-cli, /data, network sockets, etc.).
  capability,
  file,
  network,
  signal,

  # Targeted denies — known-portable AppArmor capabilities only.
  # A WebDAV bridge has no business loading kernel modules, doing raw I/O,
  # asking for sys_admin, or mounting filesystems.
  deny capability sys_module,
  deny capability sys_rawio,
  deny capability sys_admin,
  deny mount,

  # Restrict ptrace to within this profile (no cross-container snooping).
  ptrace (read,trace) peer=@{profile_name},
}
